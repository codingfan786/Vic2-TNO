			province_id = 553
			province_id = 557
			province_id = 551
			province_id = 550

		}

		NOT = {
			BGE = {
				check_variable = {
					which = bormann_protests
					value = 12
				}
			}
		}

		NOT = {has_province_modifier = student_protests}
	}

	mean_time_to_happen = {
		months = 24
	}

	immediate = {
		BGE = {
			change_variable = {
				which = bormann_protests
				value = 1
			}
		}

		add_province_modifier = {
			name = student_protests
			duration = 365
		}

		any_pop = {
			militancy = 3
			ideology = {
				factor = 2
				value = communist
			}
		}
	}

		name = "Protests have spread to $PROVINCENAME$..."
	}
