					}
				}

						}

							has_pop_culture = afro_american
							OR = {
								type = farmers
								type = labourers
							}
							militancy = 2
						}
					}
				}
			}
		}

		modifier = {
			factor = 0.5
			OR = {
				any_pop = {
					has_pop_culture = afro_american

						type = farmers

				}

						has_province_modifier = racial_integration

					}

						has_pop_culture = afro_american
						OR = {
							type = farmers
							type = labourers
						}
						militancy = 7
					}
				}
			}
		}
		modifier = {
			factor = 0.5
			OR = {
				any_pop = {
					has_pop_culture = afro_american
					OR = {
						type = farmers
						type = labourers
					}
					militancy = 8
				}
				any_neighbor_province = {
					OR = {
						has_province_modifier = racial_discontent
						has_province_modifier = racial_integration

country_event = {
	#major = yes

	id = 500061
	title = "AA_reform_title"
	desc = "AA_reform_desc_alt"

	fire_only_once = yes
	
	trigger = {
		tag = USA
		NOT = {has_country_flag = civil_rights}
		NOT = {
			usa_south = {
				NOT = {has_province_modifier = racial_integration}
				NOT = {has_province_modifier = racial_discontent}
				NOT = {has_province_modifier = civil_rights_movement}
			}
		}
	}

	mean_time_to_happen = {
		months = 3
	}

	immediate = {
		set_country_flag = civil_rights

		set_variable = {
			which = civil_rights_support
			value = 2
		}
		set_variable = {
			which = civil_rights_opposition #just for debuging...
			value = 0
		}
		any_owned = {
			limit = {
				has_province_modifier = racial_integration
			}
			THIS = {
				change_variable = {
					which = civil_rights_support
					value = 1
				}
			}
		}
		any_owned = {
			limit = {
				has_province_modifier = racial_discontent
			}
			THIS = {
				change_variable = {
					which = civil_rights_support
					value = -1
				}
				change_variable = {
					which = civil_rights_opposition
					value = 1
				}
			}
		}

		random_owned = {
			limit = {
				THIS = {
					check_variable = {
						which = civil_rights_support
						value = 1
					}
				}
			}
			THIS = {add_accepted_culture = afro_american}
			THIS = {
				any_owned = {
					remove_province_modifier = racial_integration
					remove_province_modifier = racial_discontent
					remove_province_modifier = civil_rights_movement
				}
			}
		}
		random_owned = {
			limit = {
				THIS = {
					NOT = {
						check_variable = {
							which = civil_rights_support
							value = 1
						}
					}
				}
			}
			THIS = {
				#change_tag = CSA
				country_event = 500062
			}
		}
	}
	option = {
		name = "Fantastic!"
	}
}
