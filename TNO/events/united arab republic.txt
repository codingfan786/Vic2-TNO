	id = 500209
	title = "Black Gold, Red Sand."
	desc = "event500209desc"

			limit = {
				owner = {
					ai = no

						tag = IIS

				}

					NOT = {exists = yes}

				is_capital = yes
			}
			THIS = {
				government = democracy
				political_reform = universal_voting
				political_reform = party_appointed
				political_reform = yes_meeting
				political_reform = state_press
				political_reform = all_trade_unions
				political_reform = first_past_the_post
				political_reform = underground_parties
				ruling_party_ideology = socialist
				
				primary_culture = mashriqi
				add_accepted_culture = bedouin
				add_accepted_culture = misri
				#add_accepted_culture = maghrebi
				remove_accepted_culture = south_italian
				remove_accepted_culture = north_italian
				change_tag = ARA
				release_vassal = ARA
				ITA = {
					diplomatic_influence = {
						who = ARA
						value = -800
					}
				}
				
			}
		}

		any_owned = {
			limit = {
				owner = {
					OR = {
						tag = NEJ
						tag = OMA
						tag = YEM
						tag = IRQ
						tag = EGY
						tag = ABU
						tag = JOR
						tag = SYR
						tag = ISR
					}
				}
			}
			owner = {
				war_countries = {
					end_war = THIS
				}
			}
		}

		any_owned = {
			limit = {
				owner = {
					OR = {
						tag = NEJ
						tag = OMA
						tag = YEM
						tag = IRQ
						tag = EGY
						tag = ABU
						tag = JOR
						tag = SYR
						tag = ISR
					}
				}
				is_capital = yes
			}
			ARA = {inherit = THIS}
		}

		any_owned = {
			limit = {
				OR = {
					is_core = NEJ
					is_core = OMA
					is_core = YEM
					is_core = IRQ
					is_core = EGY
					is_core = ABU
					is_core = JOR
					is_core = SYR
					is_core = ISR
				}
			}
			secede_province = ARA
			#add_core = ARA
		}

		any_owned = {
			limit = {
				has_global_flag = iis_go_away
			}
			ITA = {
				inherit = IIS
			}
			clr_global_flag = iis_go_away
		}

		any_owned = {
			limit = {
				has_global_flag = igy_go_away
			}
			ITA = {
				inherit = IGY
			}

		ITA = {
			diplomatic_influence = {
				who = ARA

			}

	}
