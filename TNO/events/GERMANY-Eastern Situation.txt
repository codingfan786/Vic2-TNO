country_event = {
	id = 500161
	title = "The Fuhrer tames the Central European Reichskommissariats."
	desc = "The Fuhrer has instituted the National Socialist regime in Bohemia, Slovakia, and Poland, ensuring the security of the Reich for a thousand years to come."
	fire_only_once = yes

	trigger = {
		OR = {
			tag = SGE
			tag = HGE
			tag = BGE
		}
		has_country_flag = nes_started
		OR = {
			NOT = {exists = POL}
			POL = {vassal_of = THIS}
		}
		OR = {
			NOT = {exists = BOH}
			BOH = {vassal_of = THIS}
		}
		OR = {
			NOT = {exists = CZH}
			CZH = {vassal_of = THIS}
		}
		OR = {
			NOT = {exists = SLV}
			SLV = {vassal_of = THIS}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		set_country_flag = nes_finished

		random_owned = {
			limit = {
				owner = {
					tag = SGE
				}
			}
			SGE = {
				set_variable = {
					which = speer_reform_points
					value = 0
				}
				set_variable = {
					which = speer_reforms_made
					value = 0
				}
				any_country = {
					limit = {
						tag = CZH
						exists = yes
					}
					SGE = {
						change_variable = {
							which = speer_reform_points
							value = 2
						}
					}
				}
				any_country = {
					limit = {
						OR = {
							tag = SLV
							tag = POL
							tag = BOH
						}
						exists = yes
					}
					SGE = {
						change_variable = {
							which = speer_reform_points
							value = 1
						}
					}
				}
			}
		}
	}

	option = {
		name = "Heil The Fuhrer!"
		prestige = 20
	}
}

country_event = {
	id = 500162
	title = "The Fuhrer tames the Eastern European Reichskommissariats."
	desc = "The Fuhrer has instituted the National Socialist regime in Moscow, Ukraine, and the Baltics, ensuring the security of the Reich for a thousand years to come."
	fire_only_once = yes

	trigger = {
		OR = {
			tag = SGE
			tag = HGE
			tag = BGE
		}

		has_country_flag = fes_started

		NOT = {exists = FEU}

		OR = {
			NOT = {exists = MSC}
			MSC = {vassal_of = THIS}
			AND = {
				set_global_flag = FEU_is_done
				MSC = {ai = no}
			}
		}
		OR = {
			NOT = {exists = BLT}
			BLT = {vassal_of = THIS}
		}
		OR = {
			NOT = {exists = UKR}
			UKR = {vassal_of = THIS}
		}
	}


	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		set_country_flag = fes_finished

		random_owned = {
			limit = {
				owner = {
					tag = SGE
				}
			}
			SGE = {
				any_country = {
					limit = {
						OR = {
							tag = MSC
							tag = BLT
							tag = UKR
						}
						exists = yes
						vassal_of = THIS
					}
					SGE = {
						change_variable = {
							which = speer_reform_points
							value = 1
						}
					}
				}
			}
		}

	}

	option = {
		name = "Heil The Fuhrer!"
		prestige = 20
	}
}
